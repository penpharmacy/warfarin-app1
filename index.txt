    <input type="number" step="0.1" id="inr" />

    <label>มีเลือดออกหรือไม่:</label>
    <select id="bleeding">
      <option value="no">ไม่มี</option>
      <option value="yes">มี</option>
    </select>

    <label>ขนาดยาเดิมต่อสัปดาห์ (mg):</label>
    <input type="number" id="weeklyDose" />

    <button onclick="calculate()">คำนวณ</button>

    <div id="result"></div>

  <script src="script.js"></script>
  <script>
    if ('serviceWorker' in navigator) {
      navigator.serviceWorker.register('sw.js')
        .then(() => console.log('Service worker registered.'));
    }

    // เพิ่มฟังก์ชั่นการคำนวณ
    function calculate() {
      const inr = parseFloat(document.getElementById('inr').value);
      const bleeding = document.getElementById('bleeding').value;
      let weeklyDose = parseFloat(document.getElementById('weeklyDose').value);

      if (isNaN(inr) || isNaN(weeklyDose)) {
        alert('กรุณากรอกข้อมูลให้ครบ');
        return;
      }

      let doseRecommendation = '';
      
      // การคำนวณขนาดยา
      if (inr < 2) {
        doseRecommendation = `ขนาดยาแนะนำ: ${weeklyDose} mg ต่อสัปดาห์`;
      } else if (inr >= 2 && inr <= 3) {
        doseRecommendation = `ขนาดยาแนะนำ: ${weeklyDose * 0.9} mg ต่อสัปดาห์`;
      } else {
        doseRecommendation = `ขนาดยาแนะนำ: ${weeklyDose * 0.8} mg ต่อสัปดาห์`;
      }

      if (bleeding === 'yes') {
        doseRecommendation += ' (ควรลดขนาดยาเนื่องจากมีเลือดออก)';
      }

      // แสดงผล
      document.getElementById('result').innerHTML = doseRecommendation;
    }
  </script>
